🔁 Polimorfisme


class Hewan {
  void suara() {
    print("Hewan bersuara");
  }
}

class Kucing extends Hewan {
  @override
  void suara() {
    print("Meong");
  }
}

class Anjing extends Hewan {
  @override
  void suara() {
    print("Guk guk");
  }
}

void main() {
  Hewan h1 = Kucing();
  Hewan h2 = Anjing();

  h1.suara();
  h2.suara();
}

🧩 Abstraksi


abstract class Kendaraan {
  void jalan();
}

class Mobil extends Kendaraan {
  @override
  void jalan() {
    print("Mobil berjalan");
  }
}

class Motor extends Kendaraan {
  @override
  void jalan() {
    print("Motor berjalan");
  }
}

void main() {
  Kendaraan k1 = Mobil();
  Kendaraan k2 = Motor();

  k1.jalan();
  k2.jalan();
}
